math.pi = 3.14159265358979
math.Deg2Rad=math.pi*2/360
math.Rad2Deg=1/math.Deg2Rad

LuaValueType={}


local function isFun(name)
	local c=sub(name,1,1)
	if c>='A' and c<='Z' then
		return true
	end
	return false
end

function LuaValueType.clamp(v,min,max)
	if v<min then return min end
	if v>max then return max end
	return v
end

local function index( meta,t,k )
	local fun = rawget(meta,k)
	if isFun(k) then
		return fun
	else
		if fun then
			return fun(t)
		else
			error('Can not find '..k..debug.traceback())
		end
	end
end

function LuaValueType.indexMeta( meta )
	return function( t,k )
		return index(meta,t,k)
	end
end
